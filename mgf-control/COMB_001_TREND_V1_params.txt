## [exits_profit_target] "10 ATR Dynamic Objective"

# Calculation: entry_price ± (target_atr_multiplier × ATR[signal_bar])

# Parameter: target_atr_multiplier
# Description: Multiplier for ATR-based profit target
# Default: 10.0 (per Mogalef PDF recommendation)
# Optimization range: [5.0, 7.0, 10.0, 15.0, 20.0] (Phase 3)
# Rationale: Mogalef DOW Jones backtest showed 10 ATR optimal
target_atr_multiplier = 10.0

# Exit logic:
#   LONG:  exit if high[i] >= entry_price + (10 × ATR[signal_bar])
#   SHORT: exit if low[i] <= entry_price - (10 × ATR[signal_bar])
# Priority: Second (after stop)

## [optimization_phases] "Independent Component Optimization Protocol"

# CRITICAL CONSTRAINT
# ==================
# Never optimize multiple components simultaneously.
# Lock all prior phase results before proceeding to next phase.
# Test each phase on Phase B unseen data to confirm robustness.

# Phase 1: Signal Optimization (10-12 hours)
# ==========================================
# Lock: Contexto, Exits, Stops at defaults
# Vary: stpmt_smooth_h, stpmt_smooth_b, stpmt_distance_max_h, stpmt_distance_max_l
# Test on: Phase A (training set), validate on Phase B (unseen)
# Decision: Pick best combo that generalizes to Phase B
# Output: Phase1_OptimizationLog.txt + best params

# Phase 2: Contexto Filters Optimization (12-16 hours)
# ===================================================
# Lock: Signal to Phase 1 best, Exits, Stops at defaults
# Vary: trend_r2, trend_r3, horaire_allowed_hours_utc, volatility_atr_min/max
# Test on: Phase A, validate on Phase B
# Decision: Filters that improve Phase B robustness without overfitting Phase A
# Output: Phase2_OptimizationLog.txt + best params

# Phase 3: Exits Optimization (6-8 hours)
# =======================================
# Lock: Signal, Contexto to Phase 1-2 best, Stops at default
# Vary: target_atr_multiplier, timescan_bars
# Test on: Phase A, validate on Phase B
# Decision: Exit logic that balances target hits vs duration discipline
# Output: Phase3_OptimizationLog.txt + best params

# Phase 4: Stops Optimization (12-14 hours)
# ==========================================
# Lock: Signal, Contexto, Exits to Phase 1-3 best
# Vary: stop_intelligent_quality, stop_intelligent_recent_volat, stop_intelligent_ref_volat, stop_intelligent_coef_volat
# Test on: Phase A, validate on Phase B
# Decision: Stop parameters that reduce average loss without choking position early
# Output: Phase4_OptimizationLog.txt + best params

# Phase 5: Final Validation
# =========================
# Combine: All Phase 1-4 best parameters
# Test: Full Phase A + Phase B data with combined params
# Robustness check: Phase B PF >= 80% × Phase A PF
# Output: Final_Validation_Report.txt + capital requirement

## [constraints]

# Constraint 1: Never test combined changes
# Reason: Cumulative optimization leads to overfitting and poor out-of-sample performance
# Enforcement: Lock all non-optimization components at defaults/prior-phase-best

# Constraint 2: Phase B is sacred (no optimization)
# Reason: Out-of-sample validation requires truly unseen data
# Enforcement: Run Phase B with Phase A params only; never optimize against Phase B results

# Constraint 3: Temporal causality maintained
# Reason: No lookahead bias, realistic execution
# Enforcement: Signal bar t → Entry bar t+1 open (always honored in backtest loop)

# Constraint 4: Minimum Phase B data
# Reason: Robust statistical validation
# Enforcement: Phase B >= 3 months unseen data from same market regime
